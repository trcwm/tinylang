# tinylang grammar

statements ::= statement [statements]

statement  ::= assignment | for_statement | var_decl

assignment    ::= VARIABLE '=' expr
for_statement ::= 'FOR' VARIABLE '=' expr 'TO' expr
var_decl      ::= 'VAR' VARIABLE { ',' VARIABLE}

expr ::= term { ('+'|'-') term }
term ::= factor { ('*'|'/') factor }
factor ::= '(' ')' | NUMBER | VARIABLE | '-' NUMBER
